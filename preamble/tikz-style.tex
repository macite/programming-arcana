\newcommand{\pseudocode}[3]
{
  \lstset{language=Pascal}
  \pseudosection{\lstinputlisting[caption={#2},label={#1}]{#3}}
}


\newcommand{\lpseudocode}[4]
{
  \lstset{language=Pascal}
  \lpseudosection{\lstinputlisting[caption={#2},label={#1},multicols=#4]{#3}}
}
